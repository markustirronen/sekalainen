1. sudo apt-get git						# Asennetaan

2. nano foo.txt							# Kyseinen tiedosto joka halutaan committaa 

3. git init							# Luo .git hakemiston

4. git add foo.txt						# Heitetään foo.txt tiedosto ".git" -hakemistoon

5. git config --global user.email markustirronen@gmail.com		# Määritellään tunnistautuminen
   git config --global user.name markustirronen

6. git commit -m "kotitehtävä 1."				# Commit nimi on "kotitehtävä 1."

7. git push -u origin master 					# Tämä käy heittämässä "master"-nimiseen branchiin
